## 5. CodeWalkthrough — 33-line TS block, 4 steps (Astro; no client directive)

<CodeWalkthrough
  lang="ts"
  caption="A tiny rate limiter, step by step"
  steps={[
    { lines: [1, 6],   note: "The bucket holds tokens; capacity and refill rate are fixed at construction." },
    { lines: [8, 15],  note: "refill() lazily adds tokens based on elapsed wall-clock time — no timer needed." },
    { lines: [17, 27], note: "take() first refills, then spends a token if one is available, returning success." },
    { lines: [29, 33], note: "Usage: construct once, then gate each request through take()." }
  ]}
  code={`class TokenBucket {
  capacity: number;
  tokens: number;
  refillPerSec: number;
  last: number;

  constructor(capacity: number, refillPerSec: number) {
    this.capacity = capacity;
    this.tokens = capacity;
    this.refillPerSec = refillPerSec;
    this.last = Date.now();
  }

  private refill() {
    const now = Date.now();
    const elapsed = (now - this.last) / 1000;
    this.tokens = Math.min(
      this.capacity,
      this.tokens + elapsed * this.refillPerSec,
    );
    this.last = now;
  }

  take(): boolean {
    this.refill();
    if (this.tokens >= 1) {
      this.tokens -= 1;
      return true;
    }
    return false;
  }
}

const limiter = new TokenBucket(10, 2);
export const allow = () => limiter.take();`}
/>
